Configuration

HTML Attributes

mm-tool="infinite-slider"
mm-slider-autoplay="3000"
mm-slider-gap="20"
mm-slider-duration="800"

Element Selectors

mm-slider-element="track"
mm-slider-element="item"
mm-slider-element="prev"
mm-slider-element="next"

Usage Example

<!-- Include the script -->
<script src="https://tools.monkeyminds.io/v1-0/slider/infinite.js"></script>

<!-- Basic slider setup -->
<div mm-tool="infinite-slider" mm-slider-autoplay="3000">
  <button mm-slider-element="prev">←</button>
  <button mm-slider-element="next">→</button>
  
  <div mm-slider-element="track">
    <div mm-slider-element="item">Slide 1</div>
    <div mm-slider-element="item">Slide 2</div>
    <div mm-slider-element="item">Slide 3</div>
  </div>
</div>
